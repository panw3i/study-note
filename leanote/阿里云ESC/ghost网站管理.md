### 重启nginx
```
[root@localhost ~]# /etc/init.d/nginx start 
[root@localhost ~]# /etc/init.d/nginx stop 
[root@localhost ~]# /etc/init.d/nginx restart 
[root@localhost ~]# /etc/init.d/nginx reload
```
